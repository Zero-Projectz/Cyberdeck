<script>

    const soul = "Je bent Cyberdeck AI. Kort, krachtig, technisch.";

    let history = JSON.parse(localStorage.getItem('cd-chats')) || [{role:"system", content:soul}];

    let engine;


    window.onload = () => {

        const t = JSON.parse(localStorage.getItem('cd-theme')) || {};

        Object.keys(t).forEach(p => document.documentElement.style.setProperty(p, t[p]));

        render();

    };


    function updateColor(p, v) {

        document.documentElement.style.setProperty(p, v);

        const t = JSON.parse(localStorage.getItem('cd-theme')) || {};

        t[p] = v; localStorage.setItem('cd-theme', JSON.stringify(t));

    }


    function toggleSettings() { const s = document.getElementById('settings'); s.style.display = s.style.display === 'none' ? 'flex' : 'none'; }

    function resetTheme() { localStorage.removeItem('cd-theme'); location.reload(); }


    function render() {

        const cont = document.getElementById('chat-container'); cont.innerHTML = '';

        history.forEach(m => { if(m.role !== 'system') append(m.role, m.content, false); });

    }


    function append(role, text, save=true) {

        const d = document.createElement('div'); d.className = `bubble ${role}`; d.innerText = text;

        document.getElementById('chat-container').appendChild(d);

        document.getElementById('chat-container').scrollTop = 99999;

        if(save) { history.push({role, content: text}); localStorage.setItem('cd-chats', JSON.stringify(history)); }

        return d;

    }


    async function sendMessage() {

        const i = document.getElementById('user-input'); const val = i.value; if(!val) return;

        append('user', val); i.value = '';

        if(!engine) {

            document.getElementById('status').innerText = "LOADING_LOCAL_LLM_GPU...";

            engine = new webllm.MLCEngine();

            await engine.reload("SmolLM-135B-Instruct-v0.2-q4f16_1-MLC");

        }

        const b = append('ai', '...');

        const res = await engine.chat.completions.create({ messages: history, stream: true });

        let full = "";

        for await (const chunk of res) { full += chunk.choices[0]?.delta?.content || ""; b.innerText = full; }

        history.push({role: "assistant", content: full});

        localStorage.setItem('cd-chats', JSON.stringify(history));

        document.getElementById('status').innerText = "CORE_ONLINE";

    }


    function exportBackup() {

        const data = { history, theme: localStorage.getItem('cd-theme') };

        const blob = new Blob([JSON.stringify(data)], {type: "application/json"});

        const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = "cyberdeck.json"; a.click();

    }
